# Обработка решения админа (Подтвердить)

        await callback.answer("Пользователь не найден.")

# Обработка решения админа (Отклонить)

# Обработка ссылок и скачивание видео
@dp.message(F.text.contains("http://") | F.text.contains("https://"))
async def handle_video_link(message: Message):
    url = message.text.strip()
    user_id = message.from_user.id
    is_vip = users_db.get(user_id, {}).get("is_vip", False)

    # Пример бизнес-логики: не-VIP могут качать, но, например, с ограничением (или только VIP)
    # В данном примере просто качаем всем, но VIP получает приветствие
    status_msg = await message.answer("📥 Начинаю скачивание видео, подождите...")

    # Настройка папки для сохранения
    output_filename = f"video_{user_id}.mp4"
    
    # Команда для yt-dlp (выбирает лучшее качество, но не более 1080p для экономии ресурсов)
    # Также качает видео сразу в один файл mp4
    cmd = f'yt-dlp -f "best[ext=mp4]/best" -o "{output_filename}" "{url}"'
    
    try:
        # Запуск скачивания в отдельном потоке, чтобы бот не зависал
        proc = await asyncio.create_subprocess_shell(cmd)
        await proc.communicate()

        if os.path.exists(output_filename):
            await status_msg.edit_text("📤 Отправляю видео...")
            # Отправка видео пользователю
            await message.reply_video(video=open(output_filename, 'rb'), caption="Вот ваше видео! ✨")
            os.remove(output_filename) # Удаляем файл с сервера после отправки
            await status_msg.delete()
        else:
            await status_msg.edit_text("❌ Не удалось скачать видео. Возможно, платформа не поддерживается или ссылка недействительна.")
            
    except Exception as e:
        await status_msg.edit_text(f"⚠️ Произошла ошибка при обработке.")
        if os.path.exists(output_filename):
            os.remove(output_filename)
